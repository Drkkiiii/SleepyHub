-- TP function

function getCurrentPlayerPOS() -- get player position
    local plyr = game.Players.LocalPlayer;
    if plyr.Character then
        return plyr.Character.HumainoidRootPart.Position;
    end
        return false;
end

function teleportTO(placeCFrame) -- teleport to place
        local plyr = game.Players.LocalPlayer;
    if plyr.Character then
        plyr.Character.HumanoidRootPart.CFrame = placeCFrame;
    end
end

function teleportToKato() -- teleport to place
        teleportTO(game:GetService("Workspace").Game.Players["Kiyosumi Kato"].HumanoidRootPart.CFrame)
end

function teleportToYasha() -- teleport to place
    teleportTO(game:GetService("Workspace").Game.Players["Yasha Ape"].HumanoidRootPart.CFrame)
end

function teleportToSThug() -- teleport to place

end

function doSpecifiedKillAura()
    spawn(function()

end
